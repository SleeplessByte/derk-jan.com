import Ingredient from '../../components/Recipes/Ingredient';
import IngredientReference from '../../components/Recipes/IngredientReference';

1. Heat <Ingredient volume={8 * 15} unit="ml" simple:amount={8} simple:unit="tablespoon">water</Ingredient> and mix together with <Ingredient weight={2 * 13} unit="g" simple:amount={2} simple:unit="tablespoon">sugar</Ingredient>, <Ingredient volume={4 * 15} unit="ml" simple:amount={4} simple:unit="tablespoon">white wine vinegar</Ingredient>, <Ingredient>salt</Ingredient> and <Ingredient>pepper</Ingredient>.
2. Let it cool down to room temperature.
3. Cut <Ingredient count={1}>cucumber</Ingredient> using a vegetable peeler or cheese slicer into long slices.
4. Cut <Ingredient count={0.5}>red onion</Ingredient> into circular or half circular slices.
5. Cut <Ingredient count={0.5}>red pepper</Ingredient> into small pieces.
6. Add the <IngredientReference>cucumber</IngredientReference> together with <Ingredient weight={1 * 5} unit="g" simple:amount={1} simple:unit="tablespoon">cloves</Ingredient>,
    the <IngredientReference>red onion</IngredientReference>, and the <IngredientReference>red pepper</IngredientReference> to the cooled down liquid.
7. Let it rest. The longer the stronger!
